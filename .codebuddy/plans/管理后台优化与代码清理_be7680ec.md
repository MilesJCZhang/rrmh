#### 3.2 后端API实现

**文件**：`routes_config.js`（修改）

**新增端点**：

1. `GET /v1/config/map` - 获取所有配置（供小程序使用）
2. `GET /v1/admin/config` - 获取配置列表（管理后台用）
3. `PUT /v1/admin/config/:key` - 更新配置（管理后台用）
4. `POST /v1/admin/config/upload` - 上传配置图片（管理后台用）

**实现逻辑**：

```javascript
// GET /v1/config/map - 获取所有配置
router.get('/map', (req, res) => {
  const db = req.app.get('db');
  const configs = db.prepare('SELECT `key`, `value` FROM system_config').all();
  const data = {};
  configs.forEach(c => {
    try {
      data[c.key] = JSON.parse(c.value);
    } catch (e) {
      data[c.key] = c.value;
    }
  });
  res.json({ code: 0, data });
});

// PUT /v1/admin/config/:key - 更新配置（需要管理员权限）
router.put('/:key', requireAuth, requireAdmin, (req, res) => {
  const db = req.app.get('db');
  const { key } = req.params;
  const { value } = req.body;
  
  const existing = db.prepare('SELECT `key` FROM system_config WHERE `key` = ?').get(key);
  if (existing) {
    db.prepare('UPDATE system_config SET `value` = ?, `updated_at` = CURRENT_TIMESTAMP WHERE `key` = ?').run(value, key);
  } else {
    db.prepare('INSERT INTO system_config (`key`, `value`) VALUES (?, ?)').run(key, value);
  }
  
  res.json({ code: 0, message: '配置更新成功' });
});
```

#### 3.3 前端服务层

**文件**：`admin-panel/src/services/config.service.ts`（新建）

**功能**：

```typescript
import request from '../utils/axios.config';

export interface ConfigItem {
  key: string;
  value: string;
  description: string;
  updated_at: string;
}

export const getConfig = async (): Promise<Record<string, any>> => {
  const response = await request.get('/v1/config/map');
  return response.data || {};
};

export const getAdminConfig = async (): Promise<ConfigItem[]> => {
  const response = await request.get('/v1/admin/config');
  return response.data || [];
};

export const updateConfig = async (key: string, value: string): Promise<void> => {
  await request.put(`/v1/admin/config/${key}`, { value });
};

export const uploadConfigImage = async (file: File): Promise<string> => {
  const formData = new FormData();
  formData.append('file', file);
  const response = await request.post('/v1/admin/config/upload', formData, {
    headers: { 'Content-Type': 'multipart/form-data' }
  });
  return response.data.url;
};
```
